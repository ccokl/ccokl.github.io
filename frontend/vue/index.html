// 挂载函数执行 mountComponent 函数
Vue.prototype.$mount = function (
  el?: string | Element,
  hydrating?: boolean
): Component {
  el = el && inBrowser ? query(el) : undefined
  return mountComponent(this, el, hydrating)
}

// mountComponent 函数内部，实例了 RenderWatcher，初始化的时候会执行回调函数，当 vm 实例中的监测的数据发生变化的时候执行回调函数
new Watcher(vm, updateComponent, noop, {
    before: function before () {
      if (vm._isMounted && !vm._isDestroyed) {
        callHook(vm, 'beforeUpdate');
      }
    }
}, true /* isRenderWatcher */);

// RenderWatcher内部的回调函数，_render函数创建虚拟DOM，_update会对比新旧虚拟DOM，生成真实DOM
updateComponent = function () {
  vm._update(vm._render(), hydrating);
};

import { reactive, ref, computed, watch, watchEffect, onMounted } from 'vue'

export default {
  setup() {
    const state = reactive({ count: 0 })
    const count = ref(0)
    const plusOne = computed(() => state.count +1 )
    // sorce 必须是响应式对象
    watch([count,plusOne],([count, plusOne], [oldCount,oldplusOne])=>{
    })
    watchEffect(() => console.log(state.count))
    onMounted(() => console.log('mounted'))
    
    //模版上下文
    return {
      state,
      increment: () => { state.count++ }
    }
  }
}
